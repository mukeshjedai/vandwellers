  void initState() {
    super.initState();

  }

    _scanning = false;

    unawaited(_setAndroidForegroundScanning(false));
    _adapterSub?.cancel();
    _ouiController.dispose();
    _intervalController.dispose();
    super.dispose();
  }

  Future<void> _setAndroidForegroundScanning(bool enable) async {
    if (!Platform.isAndroid) return;
    try {
      await _foregroundBleChannel.invokeMethod<void>(enable ? 'start' : 'stop');
    } catch (e, st) {
      debugPrint('Foreground service: $e\n$st');
    }
  }

  Future<bool> _ensurePermissions() async {
    if (!Platform.isAndroid) return true;

    final statuses = await [
      Permission.bluetoothScan,
      Permission.bluetoothConnect,
      Permission.locationWhenInUse,
      Permission.notification,
    ].request();

    final denied = statuses.entries.where((e) => !e.value.isGranted).toList();
    if (denied.isNotEmpty && mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Permissions required: ${denied.map((e) => e.key).join(', ')}')),
      );
    }
    return denied.isEmpty;
  }

  Future<void> _onScanPressed() async {
    if (_scanning) {
      setState(() => _scanning = false);
      await FlutterBluePlus.stopScan();
      await _setAndroidForegroundScanning(false);
      if (mounted) setState(() => _status = 'Stopped');
      return;
    }

    if (!await FlutterBluePlus.isSupported) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Bluetooth LE is not supported on this device')),
        );
      }
      return;
    }

    if (!await _ensurePermissions()) return;

    var state = await FlutterBluePlus.adapterState.first;
    if (state != BluetoothAdapterState.on) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Turn Bluetooth on, then try again')),
        );
      }
      return;
    }

    final intervalSeconds = int.tryParse(_intervalController.text.trim());
    if (intervalSeconds == null || intervalSeconds < 1) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Interval must be a positive number (seconds)')),
        );
      }
      return;
    }

    setState(() {
      _scanning = true;
      _status = 'Scanning…';
    });

    await _setAndroidForegroundScanning(true);
    try {
      await _scanLoop(intervalSeconds);
    } finally {
      await _setAndroidForegroundScanning(false);
    }
  }

  Future<void> _scanLoop(int intervalSeconds) async {
    while (_scanning) {
      final adapterState = await FlutterBluePlus.adapterState.first;
      if (adapterState != BluetoothAdapterState.on) {
        if (mounted) setState(() => _status = 'Waiting for Bluetooth…');
        await Future<void>.delayed(const Duration(seconds: 1));
        continue;
      }

      final ts = _formatTimestamp(DateTime.now());

      try {
        await FlutterBluePlus.startScan(
          timeout: const Duration(seconds: scanBurstSeconds),
          androidUsesFineLocation: true,
        );

        while (FlutterBluePlus.isScanningNow && _scanning) {
          await Future<void>.delayed(const Duration(milliseconds: 50));
        }
      } catch (e) {
        if (mounted) setState(() => _status = 'Scan error: $e');
        await Future<void>.delayed(Duration(seconds: intervalSeconds));
        continue;
      }

      if (!_scanning) break;

      final results = FlutterBluePlus.lastScanResults;
      final addresses = <String>{};
      for (final r in results) {
        addresses.add(_normalizeAddr(r.device.remoteId.str));
      }
      final sorted = addresses.toList()..sort();

      setState(() {
        _lastScanTime = ts;
        _devicesInRange = sorted.length;
        _lastAddresses = sorted;
        _status = 'Scanning…';
      });

      final oui = _ouiController.text.trim();
      for (final addr in sorted) {
        if (_matchesOui(addr, oui)) {
          await _onMatch(addr, ts);
        }
      }

      if (!_scanning) break;

      await Future<void>.delayed(Duration(seconds: intervalSeconds));
    }

    await FlutterBluePlus.stopScan();
    if (mounted) {
      setState(() {
        _scanning = false;
        _status = 'Stopped';
      });
    }
  }

  String _normalizeAddr(String s) => s.replaceAll('-', ':').trim().toUpperCase();

  bool _matchesOui(String address, String oui) {
    final a = _normalizeAddr(address);
    final o = _normalizeAddr(oui);
    if (o.isEmpty) return false;
    return a.startsWith(o);
  }

  Future<void> _onMatch(String addr, String ts) async {
    final line = ': $addr (Time: $ts)';
    await _appendLog(line);

    await OuiHitNotifications.showMatch(address: addr, time: ts);

    if (mounted) {
      setState(() {
        _eventLog.insert(0, '$ts  $addr');
        if (_eventLog.length > 200) _eventLog.removeLast();
      });
    }

    await SystemSound.play(SystemSoundType.alert);
    await HapticFeedback.heavyImpact();
  }

  Future<void> _appendLog(String event) async {
    try {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}${Platform.pathSeparator}$logFileName');
      final logged = _formatTimestamp(DateTime.now());
      await file.writeAsString('$event (Logged: $logged)\n', mode: FileMode.append);
    } catch (_) {}
  }

  String _formatTimestamp(DateTime dt) =>
      '${dt.year.toString().padLeft(4, '0')}-${dt.month.toString().padLeft(2, '0')}-${dt.day.toString().padLeft(2, '0')} '
      '${dt.hour.toString().padLeft(2, '0')}:${dt.minute.toString().padLeft(2, '0')}:${dt.second.toString().padLeft(2, '0')}';

import 'dart:io';

import 'package:flutter_local_notifications/flutter_local_notifications.dart';

/// Heads-up local notifications when an OUI match is detected (works while minimized via FG service).
abstract final class OuiHitNotifications {
  static final FlutterLocalNotificationsPlugin _plugin = FlutterLocalNotificationsPlugin();

  static const String _channelId = 'oui_hit_alerts';
  static const String _channelName = 'OUI match alerts';
  static int _seq = 1;

  static Future<void> init() async {
    if (!Platform.isAndroid) return;

    await _plugin.initialize(
      settings: InitializationSettings(
        android: AndroidInitializationSettings('@mipmap/ic_launcher'),
      ),
    );

    final android = _plugin.resolvePlatformSpecificImplementation<AndroidFlutterLocalNotificationsPlugin>();
    await android?.createNotificationChannel(
      const AndroidNotificationChannel(
        _channelId,
        _channelName,
        description: 'Banner alerts when a BLE address matches your OUI prefix',
        importance: Importance.max,
        playSound: true,
        enableVibration: true,
        showBadge: true,
      ),
    );
  }

  static Future<void> showMatch({required String address, required String time}) async {
    if (!Platform.isAndroid) return;

    final android = AndroidNotificationDetails(
      _channelId,
      _channelName,
      channelDescription: 'Banner alerts when a BLE address matches your OUI prefix',
      importance: Importance.max,
      priority: Priority.max,
      ticker: 'OUI device detected',
      visibility: NotificationVisibility.public,
      category: AndroidNotificationCategory.status,
      styleInformation: BigTextStyleInformation(
        '$address\nDetected at $time',
        contentTitle: 'OUI device detected',
      ),
      autoCancel: true,
      playSound: true,
      enableVibration: true,
    );

    final id = _seq;
    _seq = _seq >= 0x6FFFFFFF ? 1 : _seq + 1;

    await _plugin.show(
      id: id,
      title: 'OUI device detected',
      body: address,
      notificationDetails: NotificationDetails(android: android),
    );
  }
}

package com.example.bluetoothscanner.bluetooth_scanner

import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.app.Service
import android.content.Context
import android.content.Intent
import android.content.pm.ServiceInfo
import android.os.Build
import android.os.IBinder
import androidx.core.app.NotificationCompat

/**
 * Keeps the process in foreground while BLE scanning so Android does not suspend scans.
 * Actual BLE work stays in Flutter (flutter_blue_plus); this service only hosts the notification.
 */
class BleScanForegroundService : Service() {

    override fun onBind(intent: Intent?): IBinder? = null

    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        when (intent?.action) {
            ACTION_START -> startForegroundWithNotification()
            ACTION_STOP -> stopGracefully()
        }
        return START_NOT_STICKY
    }

    private fun startForegroundWithNotification() {
        ensureChannel()
        val notification = buildNotification()

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
            startForeground(
                NOTIFICATION_ID,
                notification,
                ServiceInfo.FOREGROUND_SERVICE_TYPE_CONNECTED_DEVICE,
            )
        } else {
            startForeground(NOTIFICATION_ID, notification)
        }
    }

    private fun stopGracefully() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
            stopForeground(STOP_FOREGROUND_REMOVE)
        } else {
            @Suppress("DEPRECATION")
            stopForeground(true)
        }
        stopSelf()
    }

    private fun ensureChannel() {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.O) return
        val mgr = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
        val channel = NotificationChannel(
            CHANNEL_ID,
            "BLE scanning",
            NotificationManager.IMPORTANCE_DEFAULT,
        ).apply {
            description = "Persistent status while OUI BLE scanning is active"
            setShowBadge(false)
        }
        mgr.createNotificationChannel(channel)
    }

    private fun buildNotification(): Notification {
        val launchIntent = Intent(this, MainActivity::class.java).apply {
            flags = Intent.FLAG_ACTIVITY_SINGLE_TOP or Intent.FLAG_ACTIVITY_CLEAR_TOP
        }
        val pendingFlags = PendingIntent.FLAG_UPDATE_CURRENT or
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) PendingIntent.FLAG_IMMUTABLE else 0
        val contentPi = PendingIntent.getActivity(this, 0, launchIntent, pendingFlags)

        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle("BLE OUI Watch")
            .setContentText("Scanning in background…")
            .setSmallIcon(android.R.drawable.stat_sys_data_bluetooth)
            .setOngoing(true)
            .setCategory(NotificationCompat.CATEGORY_SERVICE)
            .setContentIntent(contentPi)
            .setPriority(NotificationCompat.PRIORITY_DEFAULT)
            .setForegroundServiceBehavior(NotificationCompat.FOREGROUND_SERVICE_IMMEDIATE)
            .build()
    }

    companion object {
        const val ACTION_START = "com.example.bluetoothscanner.bluetooth_scanner.START_BLE_FG"
        const val ACTION_STOP = "com.example.bluetoothscanner.bluetooth_scanner.STOP_BLE_FG"
        private const val CHANNEL_ID = "ble_oui_watch_scan"
        private const val NOTIFICATION_ID = 1001
    }
}

package com.example.bluetoothscanner.bluetooth_scanner

import android.content.Intent
import androidx.core.content.ContextCompat
import io.flutter.embedding.android.FlutterActivity
import io.flutter.embedding.engine.FlutterEngine
import io.flutter.plugin.common.MethodChannel

class MainActivity : FlutterActivity() {

    override fun configureFlutterEngine(flutterEngine: FlutterEngine) {
        super.configureFlutterEngine(flutterEngine)
        MethodChannel(flutterEngine.dartExecutor.binaryMessenger, FG_CHANNEL).setMethodCallHandler { call, result ->
            when (call.method) {
                "start" -> {
                    ContextCompat.startForegroundService(
                        this,
                        Intent(this, BleScanForegroundService::class.java).apply {
                            action = BleScanForegroundService.ACTION_START
                        },
                    )
                    result.success(null)
                }

                "stop" -> {
                    startService(
                        Intent(this, BleScanForegroundService::class.java).apply {
                            action = BleScanForegroundService.ACTION_STOP
                        },
                    )
                    result.success(null)
                }

                else -> result.notImplemented()
            }
        }
    }

    companion object {
        private const val FG_CHANNEL = "com.example.bluetoothscanner.bluetooth_scanner/fg_ble_scan"
    }
}
